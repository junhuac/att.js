Open the JS console of your browser in this page. (cmd+option+j if you're in chrome).

Get a token from here: https://apimatrix.tfoundry.com/services/addressbook

Then do these one by one:

ab = new ATT.AddressBook("YOUR TOKEN");
// the GET
ab.getContacts(function (err, res) {console.log(res)});
// the POST
ab.addContact({name: {familyName: 'Guy', givenName: 'Some'}}, function (err, res) {console.log("the new contact id is: " + res.key)});
// the PUT
ab.updateContact("ID FROM LAST LINE", {name: {familyName: 'DUDE', givenName: 'Some'}}, function (err) {if (err) {console.log('this failed');} else {console.log('this worked!');}});
// the DELETE
ab.deleteContact("ID FROM PREVIOUS", function (err) {if (err) {console.log('this failed');} else {console.log('this worked!');}})
